# Extracts all of the links to articles in an Aggie article list
def get_aggie_articles(category, pages = 1, verbose = False):
    # Create an empty list to store the article URLs
    articleURLs = []
    
    # For each each page that the user requests, get the 15 URLs and store it into the list articleURLs
    for page in range(1, pages + 1):
        url = "https://theaggie.org/" + category + "/page/" + str(page)
        # "Clicks" the page
        opener = requests.get(url)

        if verbose:
            print opener.url
            
        # Downloads the data
        html_doc = opener.content
        html_parse = bs(html_doc, 'lxml-html') #lxml-html
        
        # Within the element h2 for class 'entry-title', get the attribute 'a href'
        entry_title = html_parse.findAll('h2', class_ = 'entry-title')
        for tag in entry_title:
            # Get the URL to the article
            tdTags = tag.find_all('a')[0]['href']
            # Add the URL to the list I created earlier to store the URLs
            articleURLs.append(tdTags)
        
        # Get the attribute 'a next page-numbers' 
        next_page = html_parse.findAll('a', class_ = 'next page-numbers')
        # At the last page, the length of next_page will be 0 because the next button will not exist, otherwise it is 1
        if len(next_page) == 0:
            break
    return articleURLs

The function above grabs all the links to the 15 articles in the given page number. It then returns all the article URLs as a string. Below is an example of how the function works.

def get_URL_details(one_URL):
    opener = requests.get(one_URL)
    
    # Downloads the data
    html_doc = opener.content
    html_parse = bs(html_doc, 'lxml-html')
    
    url = one_URL
    # Get only the text of the element 'h1' for entry-title
    title = html_parse.find_all('h1', class_ = 'entry-title')[0].text
    # Get the entire body content of the article, but only where it is in the tags <p>, which contains the article
    body = html_parse.find_all('div', {'itemprop' : 'articleBody'})[0].find_all('p')
    # Create an empty string, which will be used to store the author and text later
    author = str()
    text = str()
    
    # Start a counter, which I will use to help me find the author
    #i = 0
    
    # For each paragraoh, store it into text. 
    for content in body:
        text += content.text + '\n'
        #i += 1
    text = text.split('\n')
    
    i = 0
    
    while len(author) == 0:
        i += 1
        author = text[-i]
        
    # Select the article body to variable text, without the author
    text = '\n'.join(text[:-i])
    
    # Python 2 displays some Unicode characters as \uXXXX,
    # convert most of the to ASCII characters with the method call (on a string)
    text = text.translate({ 0x2018:0x27, 0x2019:0x27, 0x201C:0x22, 0x201D:0x22, 0x2026:0x20 })
    
    # order: url, title, text, and author
    url_info = {"url" : url 
                , "title": title 
                , "text" : text.replace('\n', '')
                , "author" : author
               }
    return url_info

The function above grabs a article's URL, title, body text and author. All this information is stored into a dictionary. I first grabbed the entire body of the article and looped through the body and grabbed only the text that were contained in a

(paragraph). This also ensures I do not grab other content that is not important such as the picture's caption. Next, I split each paragraph by a new line so that way I can easily grab the last line for the author.

I do this by starting at the bottom of the text variable and moving my way up until I find the first line. I need to do this because sometimes the line after the second to last sentence is just a new line and contains no content. Sometimes the last line is not the author, but also a new line. So starting from the bottom until I grabbed the first line with content ensured that I will grab the author because the author is always the last line of content in any article.

I then removed the author from the variable text.

    Extract simple lemmas based on tokenization and stemming

    return sorted(tuples, key = lambda x: (x[0], x[2]), reverse = True)
